# 3. Run subset benchmark:
uv run python -c "
from skyshield.eval.tracss_runner import run_tracss_screening, write_cdm_csv
result = run_tracss_screening(
    'data/tracss/AerospaceIVVDataset_20251009',
    pattern='*.ocm',
    mode='spherical',
    screening_radius_km=100.0,
    time_step_seconds=120.0,

filtered = [c for c in result.conjunctions if c.min_range <= 10.0]
write_cdm_csv(filtered, 'our_output.csv')
print(f'{len(filtered)} conjunctions in {result.elapsed_seconds:.1f}s')
"
